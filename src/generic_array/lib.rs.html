1//! This crate implements a structure that can be used as a generic array type.
2//!
3//! **Requires minimum Rust version of 1.83.0
4//!
5//! [Documentation on GH Pages](https://fizyk20.github.io/generic-array/generic_array/)
6//! may be required to view certain types on foreign crates.
7//!
8//! Before Rust 1.51, arrays `[T; N]` were problematic in that they couldn't be
9//! generic with respect to the length `N`, so this wouldn't work:
10//!
11//! ```compile_fail
12//! struct Foo<N> {
13//!     data: [i32; N],
14//! }
15//! ```
16//!
17//! Since 1.51, the below syntax is valid:
18//!
19//! ```rust
20//! struct Foo<const N: usize> {
21//!     data: [i32; N],
22//! }
23//! ```
24//!
25//! However, the const-generics we have as of writing this are still the minimum-viable product (`min_const_generics`), so many situations still result in errors, such as this example:
26//!
27//! ```compile_fail
28//! # struct Foo<const N: usize> {
29//! #   data: [i32; N],
30//! # }
31//! trait Bar {
32//!     const LEN: usize;
33//!
34//!     // Error: cannot perform const operation using `Self`
35//!     fn bar(&self) -> Foo<{ Self::LEN }>;
36//! }
37//! ```
38//!
39//! **generic-array** defines a new trait [`ArrayLength`] and a struct [`GenericArray<T, N: ArrayLength>`](GenericArray),
40//! which lets the above be implemented as:
41//!
42//! ```rust
43//! use generic_array::{GenericArray, ArrayLength};
44//!
45//! struct Foo<N: ArrayLength> {
46//!     data: GenericArray<i32, N>
47//! }
48//!
49//! trait Bar {
50//!     type LEN: ArrayLength;
51//!     fn bar(&self) -> Foo<Self::LEN>;
52//! }
53//! ```
54//!
55//! The [`ArrayLength`] trait is implemented for
56//! [unsigned integer types](typenum::Unsigned) from
57//! [typenum]. For example, [`GenericArray<T, U5>`] would work almost like `[T; 5]`:
58//!
59//! ```rust
60//! # use generic_array::{ArrayLength, GenericArray};
61//! use generic_array::typenum::U5;
62//!
63//! struct Foo<T, N: ArrayLength> {
64//!     data: GenericArray<T, N>
65//! }
66//!
67//! let foo = Foo::<i32, U5> { data: GenericArray::default() };
68//! ```
69//!
70//! The `arr!` macro is provided to allow easier creation of literal arrays, as shown below:
71//!
72//! ```rust
73//! # use generic_array::arr;
74//! let array = arr![1, 2, 3];
75//! //  array: GenericArray<i32, typenum::U3>
76//! assert_eq!(array[2], 3);
77//! ```
78//! ## Feature flags
79//!
80//! ```toml
81//! [dependencies.generic-array]
82//! features = [
83//!     "serde",         # Serialize/Deserialize implementation
84//!     "zeroize",       # Zeroize implementation for setting array elements to zero
85//!     "const-default", # Compile-time const default value support via trait
86//!     "alloc",         # Enables From/TryFrom implementations between GenericArray and Vec<T>/Box<[T]>
87//!     "faster-hex"     # Enables internal use of the `faster-hex` crate for faster hex encoding via SIMD
88//! ]
89//! ```

155/// `ArrayLength` is a type-level [`Unsigned`] integer used to
156/// define the number of elements in a [`GenericArray`].
157///
158/// Consider `N: ArrayLength` to be equivalent to `const N: usize`
159///
160/// ```
161/// # use generic_array::{GenericArray, ArrayLength};
162/// fn foo<N: ArrayLength>(arr: GenericArray<i32, N>) -> i32 {
163///     arr.iter().sum()
164/// }
165/// ```
166/// is equivalent to:
167/// ```
168/// fn foo<const N: usize>(arr: [i32; N]) -> i32 {
169///     arr.iter().sum()
170/// }
171/// ```
172///
173/// # Safety
174///
175/// This trait is effectively sealed due to only being allowed on [`Unsigned`] types,
176/// and therefore cannot be implemented in user code.
177pub unsafe trait ArrayLength: Unsigned + 'static {
178    /// Associated type representing the underlying contiguous memory
179    /// that constitutes an array with the given number of elements.
180    ///
181    /// This is an implementation detail, but is required to be public in cases where certain attributes
182    /// of the inner type of [`GenericArray`] cannot be proven, such as [`Copy`] bounds.
183    ///
184    /// [`Copy`] example:
185    /// ```
186    /// # use generic_array::{GenericArray, ArrayLength};
187    /// struct MyType<N: ArrayLength> {
188    ///     data: GenericArray<f32, N>,
189    /// }
190    ///
191    /// impl<N: ArrayLength> Clone for MyType<N> where N::ArrayType<f32>: Copy {
192    ///     fn clone(&self) -> Self { MyType { ..*self } }
193    /// }
194    ///
195    /// impl<N: ArrayLength> Copy for MyType<N> where N::ArrayType<f32>: Copy {}
196    /// ```
197    ///
198    /// Alternatively, using the entire `GenericArray<f32, N>` type as the bounds works:
199    /// ```ignore
200    /// where GenericArray<f32, N>: Copy
201    /// ```
202    type ArrayType<T>: Sealed;
203}
204
205unsafe impl ArrayLength for UTerm {
206    #[doc(hidden)]
207    type ArrayType<T> = [T; 0];
208}

210/// Implemented for types which can have an associated [`ArrayLength`],
211/// such as [`Const<N>`] for use with const-generics.
212///
213/// ```
214/// use generic_array::{GenericArray, IntoArrayLength, ConstArrayLength, typenum::Const};
215///
216/// fn some_array_interopt<const N: usize>(value: [u32; N]) -> GenericArray<u32, ConstArrayLength<N>>
217/// where
218///     Const<N>: IntoArrayLength,
219/// {
220///     let ga = GenericArray::from(value);
221///     // do stuff
222///     ga
223/// }
224/// ```
225///
226/// This is mostly to simplify the `where` bounds, equivalent to:
227///
228/// ```
229/// use generic_array::{GenericArray, ArrayLength, typenum::{Const, U, ToUInt}};
230///
231/// fn some_array_interopt<const N: usize>(value: [u32; N]) -> GenericArray<u32, U<N>>
232/// where
233///     Const<N>: ToUInt,
234///     U<N>: ArrayLength,
235/// {
236///     let ga = GenericArray::from(value);
237///     // do stuff
238///     ga
239/// }
240/// ```
241pub trait IntoArrayLength {
242    /// The associated `ArrayLength`
243    type ArrayLength: ArrayLength;
244}

318/// Struct representing a generic array - `GenericArray<T, N>` works like `[T; N]`
319///
320/// For how to implement [`Copy`] on structs using a generic-length `GenericArray` internally, see
321/// the docs for [`ArrayLength::ArrayType`].
322///
323/// # Usage Notes
324///
325/// ### Initialization
326///
327/// Initialization of known-length `GenericArray`s can be done via the [`arr![]`](arr!) macro,
328/// or [`from_array`](GenericArray::from_array)/[`from_slice`](GenericArray::from_slice).
329///
330/// For generic arrays of unknown/generic length, several safe methods are included to initialize
331/// them, such as the [`GenericSequence::generate`] method:
332///
333/// ```rust
334/// use generic_array::{GenericArray, sequence::GenericSequence, typenum, arr};
335///
336/// let evens: GenericArray<i32, typenum::U4> =
337///            GenericArray::generate(|i: usize| i as i32 * 2);
338///
339/// assert_eq!(evens, arr![0, 2, 4, 6]);
340/// ```
341///
342/// Furthermore, [`FromIterator`] and [`try_from_iter`](GenericArray::try_from_iter) exist to construct them
343/// from iterators, but will panic/fail if not given exactly the correct number of elements.
344///
345/// ### Utilities
346///
347/// The [`GenericSequence`], [`FunctionalSequence`], [`Lengthen`], [`Shorten`], [`Split`], and [`Concat`] traits implement
348/// some common operations on generic arrays.
349///
350/// ### Optimizations
351///
352/// Prefer to use the slice iterators like `.iter()`/`.iter_mut()` rather than by-value [`IntoIterator`]/[`GenericArrayIter`] if you can.
353/// Slices optimize better. Using the [`FunctionalSequence`] methods also optimize well.
354///
355/// # How it works
356///
357/// The `typenum` crate uses Rust's type system to define binary integers as nested types,
358/// and allows for operations which can be applied to those type-numbers, such as `Add`, `Sub`, etc.
359///
360/// e.g. `6` would be `UInt<UInt<UInt<UTerm, B1>, B1>, B0>`
361///
362/// `generic-array` uses this nested type to recursively allocate contiguous elements, statically.
363/// The [`ArrayLength`] trait is implemented on `UInt<N, B0>`, `UInt<N, B1>` and `UTerm`,
364/// which correspond to even, odd and zero numeric values, respectively.
365/// Together, these three cover all cases of `Unsigned` integers from `typenum`.
366/// For `UInt<N, B0>` and `UInt<N, B1>`, it peels away the highest binary digit and
367/// builds up a recursive structure that looks almost like a binary tree.
368/// Then, within `GenericArray`, the recursive structure is reinterpreted as a contiguous
369/// chunk of memory and allowing access to it as a slice.
370///
371/// <details>
372/// <summary><strong>Expand for internal structure demonstration</strong></summary>
373///
374/// For example, `GenericArray<T, U6>` more or less expands to (at compile time):
375///
376/// ```ignore
377/// GenericArray {
378///     // 6 = UInt<UInt<UInt<UTerm, B1>, B1>, B0>
379///     data: EvenData {
380///         // 3 = UInt<UInt<UTerm, B1>, B1>
381///         left: OddData {
382///             // 1 = UInt<UTerm, B1>
383///             left: OddData {
384///                 left: (),  // UTerm
385///                 right: (), // UTerm
386///                 data: T,   // Element 0
387///             },
388///             // 1 = UInt<UTerm, B1>
389///             right: OddData {
390///                 left: (),  // UTerm
391///                 right: (), // UTerm
392///                 data: T,   // Element 1
393///             },
394///             data: T        // Element 2
395///         },
396///         // 3 = UInt<UInt<UTerm, B1>, B1>
397///         right: OddData {
398///             // 1 = UInt<UTerm, B1>
399///             left: OddData {
400///                 left: (),  // UTerm
401///                 right: (), // UTerm
402///                 data: T,   // Element 3
403///             },
404///             // 1 = UInt<UTerm, B1>
405///             right: OddData {
406///                 left: (),  // UTerm
407///                 right: (), // UTerm
408///                 data: T,   // Element 4
409///             },
410///             data: T        // Element 5
411///         }
412///     }
413/// }
414/// ```
415///
416/// This has the added benefit of only being `log2(N)` deep, which is important for things like `Drop`
417/// to avoid stack overflows, since we can't implement `Drop` manually.
418///
419/// Then, we take the contiguous block of data and cast it to `*const T` or `*mut T` and use it as a slice:
420///
421/// ```ignore
422/// unsafe {
423///     slice::from_raw_parts(
424///         self as *const GenericArray<T, N> as *const T,
425///         <N as Unsigned>::USIZE
426///     )
427/// }
428/// ```
429///
430/// </details>
431#[repr(transparent)]
432pub struct GenericArray<T, N: ArrayLength> {
433    #[allow(dead_code)] // data is never accessed directly
434    data: N::ArrayType<T>,
435}

474impl<T, N: ArrayLength> FromIterator<T> for GenericArray<T, N> {
475    /// Create a `GenericArray` from an iterator.
476    ///
477    /// Will panic if the number of elements is not exactly the array length.
478    ///
479    /// See [`GenericArray::try_from_iter]` for a fallible alternative.
480    #[inline]
481    fn from_iter<I>(iter: I) -> GenericArray<T, N>
482    where
483        I: IntoIterator<Item = T>,
484    {
485        match Self::try_from_iter(iter) {
486            Ok(res) => res,
487            Err(_) => from_iter_length_fail(N::USIZE),
488        }
489    }
490}

498unsafe impl<T, N: ArrayLength> GenericSequence<T> for GenericArray<T, N>
499where
500    Self: IntoIterator<Item = T>,
501{
502    type Length = N;
503    type Sequence = Self;
504
505    #[inline(always)]
506    fn generate<F>(mut f: F) -> GenericArray<T, N>
507    where
508        F: FnMut(usize) -> T,
509    {
510        unsafe {
511            let mut array = GenericArray::<T, N>::uninit();
512            let mut builder = IntrusiveArrayBuilder::new(&mut array);
513
514            {
515                let (builder_iter, position) = builder.iter_position();
516
517                builder_iter.enumerate().for_each(|(i, dst)| {
518                    dst.write(f(i));
519                    *position += 1;
520                });
521            }
522
523            builder.finish();
524            IntrusiveArrayBuilder::array_assume_init(array)
525        }
526    }
527
528    #[inline(always)]
529    fn inverted_zip<B, U, F>(
530        self,
531        lhs: GenericArray<B, Self::Length>,
532        mut f: F,
533    ) -> MappedSequence<GenericArray<B, Self::Length>, B, U>
534    where
535        GenericArray<B, Self::Length>:
536            GenericSequence<B, Length = Self::Length> + MappedGenericSequence<B, U>,
537        Self: MappedGenericSequence<T, U>,
538        F: FnMut(B, Self::Item) -> U,
539    {
540        unsafe {
541            if mem::needs_drop::<T>() || mem::needs_drop::<B>() {
542                let mut left = ArrayConsumer::new(lhs);
543                let mut right = ArrayConsumer::new(self);
544
545                let (left_array_iter, left_position) = left.iter_position();
546                let (right_array_iter, right_position) = right.iter_position();
547
548                FromIterator::from_iter(left_array_iter.zip(right_array_iter).map(|(l, r)| {
549                    let left_value = ptr::read(l);
550                    let right_value = ptr::read(r);
551
552                    *left_position += 1;
553                    *right_position = *left_position;
554
555                    f(left_value, right_value)
556                }))
557            } else {
558                // Despite neither needing `Drop`, they may not be `Copy`, so be paranoid
559                // and avoid anything related to drop anyway. Assume it's moved out on each read.
560                let left = ManuallyDrop::new(lhs);
561                let right = ManuallyDrop::new(self);
562
563                // Neither right nor left require `Drop` be called, so choose an iterator that's easily optimized
564                //
565                // Note that because ArrayConsumer checks for `needs_drop` itself, if `f` panics then nothing
566                // would have been done about it anyway. Only the other branch needs `ArrayConsumer`
567                FromIterator::from_iter(left.iter().zip(right.iter()).map(|(l, r)| {
568                    f(ptr::read(l), ptr::read(r)) //
569                }))
570            }
571        }
572    }
573
574    #[inline(always)]
575    fn inverted_zip2<B, Lhs, U, F>(self, lhs: Lhs, mut f: F) -> MappedSequence<Lhs, B, U>
576    where
577        Lhs: GenericSequence<B, Length = Self::Length> + MappedGenericSequence<B, U>,
578        Self: MappedGenericSequence<T, U>,
579        F: FnMut(Lhs::Item, Self::Item) -> U,
580    {
581        unsafe {
582            if mem::needs_drop::<T>() {
583                let mut right = ArrayConsumer::new(self);
584
585                let (right_array_iter, right_position) = right.iter_position();
586
587                FromIterator::from_iter(right_array_iter.zip(lhs).map(|(r, left_value)| {
588                    let right_value = ptr::read(r);
589
590                    *right_position += 1;
591
592                    f(left_value, right_value)
593                }))
594            } else {
595                let right = ManuallyDrop::new(self);
596
597                // Similar logic to `inverted_zip`'s no-drop branch
598                FromIterator::from_iter(right.iter().zip(lhs).map(|(r, left_value)| {
599                    f(left_value, ptr::read(r)) //
600                }))
601            }
602        }
603    }
604}

613impl<T, N: ArrayLength> FunctionalSequence<T> for GenericArray<T, N>
614where
615    Self: GenericSequence<T, Item = T, Length = N>,
616{
617    #[inline(always)]
618    fn map<U, F>(self, mut f: F) -> MappedSequence<Self, T, U>
619    where
620        Self: MappedGenericSequence<T, U>,
621        F: FnMut(T) -> U,
622    {
623        unsafe {
624            let mut source = ArrayConsumer::new(self);
625
626            let (array_iter, position) = source.iter_position();
627
628            FromIterator::from_iter(array_iter.map(|src| {
629                let value = ptr::read(src);
630
631                *position += 1;
632
633                f(value)
634            }))
635        }
636    }
637
638    #[inline(always)]
639    fn zip<B, Rhs, U, F>(self, rhs: Rhs, f: F) -> MappedSequence<Self, T, U>
640    where
641        Self: MappedGenericSequence<T, U>,
642        Rhs: MappedGenericSequence<B, U, Mapped = MappedSequence<Self, T, U>>,
643        Rhs: GenericSequence<B, Length = Self::Length>,
644        F: FnMut(T, Rhs::Item) -> U,
645    {
646        rhs.inverted_zip(self, f)
647    }
648
649    #[inline(always)]
650    fn fold<U, F>(self, init: U, mut f: F) -> U
651    where
652        F: FnMut(U, T) -> U,
653    {
654        unsafe {
655            let mut source = ArrayConsumer::new(self);
656
657            let (array_iter, position) = source.iter_position();
658
659            array_iter.fold(init, |acc, src| {
660                let value = ptr::read(src);
661                *position += 1;
662                f(acc, value)
663            })
664        }
665    }
666}

691    ///
692    /// # Panics
693    ///
694    /// Panics if the slice is not equal to the length of the array.
695    ///
696    /// Consider [`TryFrom`]/[`TryInto`] for a fallible conversion,
697    /// or [`try_from_slice`](GenericArray::try_from_slice) for use in const expressions.

751    /// Converts a slice of `T` elements into a slice of `GenericArray<T, N>` chunks.
752    ///
753    /// Any remaining elements that do not fill the array will be returned as a second slice.
754    ///
755    /// # Panics
756    ///
757    /// Panics if `N` is `U0` _AND_ the input slice is not empty.
758    pub const fn chunks_from_slice(slice: &[T]) -> (&[GenericArray<T, N>], &[T]) {
759        if N::USIZE == 0 {
760            assert!(slice.is_empty(), "GenericArray length N must be non-zero");
761            return (&[], &[]);
762        }
763
764        // NOTE: Using `slice.split_at` adds an unnecessary assert
765        let num_chunks = slice.len() / N::USIZE; // integer division
766        let num_in_chunks = num_chunks * N::USIZE;
767        let num_remainder = slice.len() - num_in_chunks;
768
769        unsafe {
770            (
771                slice::from_raw_parts(slice.as_ptr() as *const GenericArray<T, N>, num_chunks),
772                slice::from_raw_parts(slice.as_ptr().add(num_in_chunks), num_remainder),
773            )
774        }
775    }

784    pub const fn chunks_from_slice_mut(slice: &mut [T]) -> (&mut [GenericArray<T, N>], &mut [T]) {

786            assert!(slice.is_empty(), "GenericArray length N must be non-zero");
787            return (&mut [], &mut []);
788        }
789
790        // NOTE: Using `slice.split_at_mut` adds an unnecessary assert

793        let num_remainder = slice.len() - num_in_chunks;
794
795        unsafe {
796            (

799                    num_chunks,
800                ),
801                slice::from_raw_parts_mut(slice.as_mut_ptr().add(num_in_chunks), num_remainder),
802            )
803        }
804    }

891    ///
892    /// # Safety
893    ///
894    /// It is up to the caller to guarantee that all elements of the array are in an initialized state.
895    ///
896    /// # Example
897    ///
898    /// ```
899    /// # use core::mem::MaybeUninit;
900    /// # use generic_array::{GenericArray, typenum::U3, arr};
901    /// let mut array: GenericArray<MaybeUninit<i32>, U3> = GenericArray::uninit();
902    /// array[0].write(0);
903    /// array[1].write(1);
904    /// array[2].write(2);

906    /// // SAFETY: Now safe as we initialised all elements
907    /// let array = unsafe {

909    /// };
910    ///

915        const_transmute::<_, MaybeUninit<GenericArray<T, N>>>(array).assume_init()
916    }

949impl<T, N: ArrayLength> GenericArray<T, N> {
950    /// Fallible equivalent of [`FromIterator::from_iter`]
951    ///
952    /// Given iterator must yield exactly `N` elements or an error will be returned. Using [`.take(N)`](Iterator::take)
953    /// with an iterator longer than the array may be helpful.
954    #[inline]
955    pub fn try_from_iter<I>(iter: I) -> Result<Self, LengthError>
956    where
957        I: IntoIterator<Item = T>,
958    {
959        let mut iter = iter.into_iter();
960
961        // pre-checks
962        match iter.size_hint() {
963            // if the lower bound is greater than N, array will overflow
964            (n, _) if n > N::USIZE => return Err(LengthError),
965            // if the upper bound is smaller than N, array cannot be filled
966            (_, Some(n)) if n < N::USIZE => return Err(LengthError),
967            _ => {}
968        }
969
970        unsafe {
971            let mut array = GenericArray::uninit();
972            let mut builder = IntrusiveArrayBuilder::new(&mut array);
973
974            builder.extend(&mut iter);
975
976            if !builder.is_full() || iter.next().is_some() {
977                return Err(LengthError);
978            }
979
980            Ok({
981                builder.finish();
982                IntrusiveArrayBuilder::array_assume_init(array)
983            })
984        }
985    }
986}

988/// A const reimplementation of the [`transmute`](core::mem::transmute) function,
989/// avoiding problems when the compiler can't prove equal sizes.
990///
991/// # Safety
992/// Treat this the same as [`transmute`](core::mem::transmute), or (preferably) don't use it at all.
993#[inline(always)]
994#[cfg_attr(not(feature = "internals"), doc(hidden))]
995pub const unsafe fn const_transmute<A, B>(a: A) -> B {
996    if mem::size_of::<A>() != mem::size_of::<B>() {
997        panic!("Size mismatch for generic_array::const_transmute");
998    }
999
1000    #[repr(C)]
1001    union Union<A, B> {
1002        a: ManuallyDrop<A>,
1003        b: ManuallyDrop<B>,
1004    }
1005
1006    let a = ManuallyDrop::new(a);
1007    ManuallyDrop::into_inner(Union { a }.b)
1008}
