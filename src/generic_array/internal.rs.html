9/// **UNSAFE**: Creates an array one element at a time using a mutable iterator of pointers.
10///
11/// You MUST increment the position while iterating to mark off created elements,
12/// which will be dropped if `into_inner` is not called.
13///
14/// This is soft-deprecated in favor of [`IntrusiveArrayBuilder`] due to Rust's
15/// lack of return-value optimization causing issues moving the array in/out of the struct.
16/// Still works fine for smaller arrays, though.
17pub struct ArrayBuilder<T, N: ArrayLength> {
18    array: GenericArray<MaybeUninit<T>, N>,
19    position: usize,
20}
21
22impl<T, N: ArrayLength> ArrayBuilder<T, N> {
23    /// Begin building an array
24    #[inline(always)]
25    pub const fn new() -> ArrayBuilder<T, N> {
26        ArrayBuilder {
27            array: GenericArray::uninit(),
28            position: 0,
29        }
30    }
31
32    /// Consume an iterator, `.zip`-ing it to fill some or all of the array. This does not check if the
33    /// iterator had extra elements or too few elements.
34    ///
35    /// This makes no attempt to continue where a previous `extend` leaves off. Therefore, it should
36    /// only be used once per `ArrayBuilder`.
37    #[inline(always)]
38    pub unsafe fn extend(&mut self, source: impl Iterator<Item = T>) {
39        let (destination, position) = (self.array.iter_mut(), &mut self.position);
40
41        destination.zip(source).for_each(|(dst, src)| {
42            dst.write(src);
43            *position += 1;
44        });
45    }
46
47    /// Returns true if the write position equals the array size
48    #[inline(always)]
49    pub const fn is_full(&self) -> bool {
50        self.position == N::USIZE
51    }
52
53    /// Creates a mutable iterator for writing to the array elements.
54    ///
55    /// You MUST increment the position value (given as a mutable reference) as you iterate
56    /// to mark how many elements have been created.
57    ///
58    /// ```
59    /// #[cfg(feature = "internals")]
60    /// # {
61    /// # use generic_array::{GenericArray, internals::ArrayBuilder, typenum::U5};
62    /// # struct SomeType;
63    /// fn make_some_struct() -> SomeType { SomeType }
64    /// unsafe {
65    ///     let mut builder = ArrayBuilder::<SomeType, U5>::new();
66    ///     let (dst_iter, position) = builder.iter_position();
67    ///     for dst in dst_iter {
68    ///         dst.write(make_some_struct());
69    ///         // MUST be done AFTER ownership of the value has been given to `dst.write`
70    ///         *position += 1;
71    ///     }
72    ///     let your_array = builder.assume_init();
73    /// }
74    /// # }
75    /// ```
76    #[inline(always)]
77    pub unsafe fn iter_position(
78        &'_ mut self,
79    ) -> (slice::IterMut<'_, MaybeUninit<T>>, &'_ mut usize) {
80        (self.array.iter_mut(), &mut self.position)
81    }
82
83    /// When done writing (assuming all elements have been written to),
84    /// get the inner array.
85    ///
86    /// This method is `const` since Rust 1.83.0, but non-`const` before.
87    #[rustversion::attr(since(1.83), const)]
88    #[inline(always)]
89    pub unsafe fn assume_init(self) -> GenericArray<T, N> {
90        debug_assert!(self.is_full());
91
92        let array = ptr::read(&self.array);
93        mem::forget(self);
94        GenericArray::assume_init(array)
95    }
96}
97
98impl<T, N: ArrayLength> Drop for ArrayBuilder<T, N> {
99    fn drop(&mut self) {
100        unsafe {
101            ptr::drop_in_place(
102                // Same cast as MaybeUninit::slice_assume_init_mut
103                self.array.get_unchecked_mut(..self.position) as *mut [MaybeUninit<T>]
104                    as *mut [T],
105            );
106        }
107    }
108}

110/// Similar to [`ArrayBuilder`] but uses a reference to a pre-allocated array, be

117impl<'a, T, N: ArrayLength> IntrusiveArrayBuilder<'a, T, N> {
118    /// Begin building an array
119    ///
120    /// This method is `const` since Rust 1.83.0, but non-`const` before.
121    #[rustversion::attr(since(1.83), const)]
122    #[inline(always)]
123    pub fn new(array: &'a mut GenericArray<MaybeUninit<T>, N>) -> IntrusiveArrayBuilder<'a, T, N> {
124        IntrusiveArrayBuilder { array, position: 0 }
125    }
126
127    /// Begin building an array from a `MaybeUninit<GenericArray<T, N>>`, which can potentially
128    /// result in better codegen.
129    ///
130    /// This method is `const` since Rust 1.83.0, but non-`const` before.
131    #[rustversion::attr(since(1.83), const)]
132    #[inline(always)]
133    pub fn new_alt(
134        array: &'a mut MaybeUninit<GenericArray<T, N>>,
135    ) -> IntrusiveArrayBuilder<'a, T, N> {
136        // SAFETY: GenericArray<MaybeUninit<T>, N> has the same memory layout as MaybeUninit<GenericArray<T, N>>
137        IntrusiveArrayBuilder::new(unsafe {
138            mem::transmute::<
139                &mut MaybeUninit<GenericArray<T, N>>,
140                &mut GenericArray<MaybeUninit<T>, N>,
141            >(array)
142        })
143    }
144
145    /// Consume an iterator, `.zip`-ing it to fill some or all of the array. This does not check if the
146    /// iterator had extra elements or too few elements.
147    ///
148    /// This makes no attempt to continue where a previous `extend` leaves off. Therefore, it should
149    /// only be used once per `ArrayBuilder`.
150    #[inline(always)]
151    pub unsafe fn extend(&mut self, source: impl Iterator<Item = T>) {
152        let (destination, position) = (self.array.iter_mut(), &mut self.position);
153
154        destination.zip(source).for_each(|(dst, src)| {
155            dst.write(src);
156            *position += 1;
157        });
158    }
159
160    /// Consume a fallible iterator, `.zip`-ing it to fill some or all of the array. This does not check if the
161    /// iterator had extra elements or too few elements.
162    ///
163    /// This makes no attempt to continue where a previous `extend` leaves off. Therefore, it should
164    /// only be used once per `ArrayBuilder`.
165    #[inline(always)]
166    pub unsafe fn try_extend<E>(
167        &mut self,
168        source: impl Iterator<Item = Result<T, E>>,
169    ) -> Result<(), E> {
170        let (destination, position) = (self.array.iter_mut(), &mut self.position);
171
172        destination
173            .zip(source)
174            .try_for_each(|(dst, src)| match src {
175                Ok(value) => {
176                    dst.write(value);
177                    *position += 1;
178                    Ok(())
179                }
180                Err(e) => Err(e),
181            })
182    }
183
184    /// Returns true if the write position equals the array size
185    #[inline(always)]
186    pub const fn is_full(&self) -> bool {
187        self.position == N::USIZE
188    }
189
190    /// Creates a mutable iterator for writing to the array elements.
191    ///
192    /// You MUST increment the position value (given as a mutable reference) as you iterate
193    /// to mark how many elements have been created.
194    ///
195    /// ```
196    /// #[cfg(feature = "internals")]
197    /// # {
198    /// # use generic_array::{GenericArray, internals::IntrusiveArrayBuilder, typenum::U5};
199    /// # use core::mem::MaybeUninit;
200    /// # struct SomeType;
201    /// fn make_some_struct() -> SomeType { SomeType }
202    /// unsafe {
203    ///     let mut array = MaybeUninit::<GenericArray<SomeType, U5>>::uninit();
204    ///     let mut builder = IntrusiveArrayBuilder::new_alt(&mut array);
205    ///     let (dst_iter, position) = builder.iter_position();
206    ///     for dst in dst_iter {
207    ///         dst.write(make_some_struct());
208    ///         // MUST be done AFTER ownership of the value has been given to `dst.write`
209    ///         *position += 1;
210    ///     }
211    ///
212    ///     let your_array = builder.finish_and_assume_init();
213    /// }
214    /// # }
215    /// ```
216    #[inline(always)]
217    pub unsafe fn iter_position(
218        &'_ mut self,
219    ) -> (slice::IterMut<'_, MaybeUninit<T>>, &'_ mut usize) {
220        (self.array.iter_mut(), &mut self.position)
221    }
222
223    /// When done writing (assuming all elements have been written to), mark the builder
224    /// as finished to avoid dropping the initialized elements.
225    #[inline(always)]
226    pub const unsafe fn finish(self) {
227        debug_assert!(self.is_full());
228        mem::forget(self)
229    }
230
231    /// When done writing (assuming all elements have been written to), mark the builder
232    /// as finished to avoid dropping the initialized elements, and get the inner array.
233    ///
234    /// This method is `const` since Rust 1.71.0, but non-`const` before.
235    #[rustversion::attr(since(1.71), const)]
236    #[inline(always)]
237    pub unsafe fn finish_and_assume_init(self) -> GenericArray<T, N> {
238        debug_assert!(self.is_full());
239
240        // Same-ish as `assume_init`/`transmute_copy` but generates better codegen in Debug builds
241        // SAFETY: The array is initialized, and by definition is the same size and alignment as GenericArray<T, N>
242        let array = ptr::read(self.array as *const _ as *const GenericArray<T, N>);
243
244        // avoid Drop logic, and since this is intrusive, doing this after reading the array is fine
245        mem::forget(self);
246
247        array
248    }
249
250    /// Similar to [`GenericArray::assume_init`] but for arrays passed to `IntrusiveArrayBuilder`.
251    #[deprecated(
252        since = "1.3.4",
253        note = "See `iter_position` for modern usage pattern without this function"
254    )]
255    #[inline(always)]
256    pub unsafe fn array_assume_init(array: GenericArray<MaybeUninit<T>, N>) -> GenericArray<T, N> {
257        ptr::read(&array as *const _ as *const MaybeUninit<GenericArray<T, N>>).assume_init()
258    }
259}

273/// **UNSAFE**: Consumes an array one element at a time.
274///
275/// You MUST increment the position while iterating and any leftover elements
276/// will be dropped if position does not go to N
277///
278/// This is soft-deprecated in favor of [`IntrusiveArrayConsumer`] due to Rust's
279/// lack of return-value optimization causing issues moving the array in/out of the struct.
280pub struct ArrayConsumer<T, N: ArrayLength> {
281    array: ManuallyDrop<GenericArray<T, N>>,
282    position: usize,
283}

285impl<T, N: ArrayLength> ArrayConsumer<T, N> {
286    /// Give ownership of the array to the consumer
287    #[inline(always)]

289        ArrayConsumer {
290            array: ManuallyDrop::new(array),
291            position: 0,
292        }
293    }
294
295    /// Creates an iterator and mutable reference to the internal position
296    /// to keep track of consumed elements.
297    ///
298    /// You MUST increment the position as you iterate to mark off consumed elements.
299    #[inline(always)]
300    pub unsafe fn iter_position(&'_ mut self) -> (slice::Iter<'_, T>, &'_ mut usize) {
301        (self.array.iter(), &mut self.position)
302    }
303}
