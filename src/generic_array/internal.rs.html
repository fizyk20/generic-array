9/// **UNSAFE**: Creates an array one element at a time using a mutable iterator of pointers.
10///
11/// You MUST increment the position while iterating to mark off created elements,
12/// which will be dropped if `into_inner` is not called.
13///
14/// This is soft-deprecated in favor of [`IntrusiveArrayBuilder`] due to Rust's
15/// lack of return-value optimization causing issues moving the array out of the struct.
16/// Still works fine for smaller arrays, though.
17pub struct ArrayBuilder<T, N: ArrayLength> {
18    array: GenericArray<MaybeUninit<T>, N>,
19    position: usize,
20}
21
22impl<T, N: ArrayLength> ArrayBuilder<T, N> {
23    /// Begin building an array
24    #[inline(always)]
25    pub const fn new() -> ArrayBuilder<T, N> {
26        ArrayBuilder {
27            array: GenericArray::uninit(),
28            position: 0,
29        }
30    }
31
32    /// Consume an iterator, `.zip`-ing it to fill some or all of the array. This does not check if the
33    /// iterator had extra elements or too few elements.
34    ///
35    /// This makes no attempt to continue where a previous `extend` leaves off. Therefore, it should
36    /// only be used once per `ArrayBuilder`.
37    #[inline(always)]
38    pub unsafe fn extend(&mut self, source: impl Iterator<Item = T>) {
39        let (destination, position) = (self.array.iter_mut(), &mut self.position);
40
41        destination.zip(source).for_each(|(dst, src)| {
42            dst.write(src);
43            *position += 1;
44        });
45    }
46
47    /// Returns true if the write position equals the array size
48    #[inline(always)]
49    pub const fn is_full(&self) -> bool {
50        self.position == N::USIZE
51    }
52
53    /// Creates a mutable iterator for writing to the array elements.
54    ///
55    /// You MUST increment the position value (given as a mutable reference) as you iterate
56    /// to mark how many elements have been created.
57    ///
58    /// ```
59    /// #[cfg(feature = "internals")]
60    /// # {
61    /// # use generic_array::{GenericArray, internals::ArrayBuilder, typenum::U5};
62    /// # struct SomeType;
63    /// fn make_some_struct() -> SomeType { SomeType }
64    /// unsafe {
65    ///     let mut builder = ArrayBuilder::<SomeType, U5>::new();
66    ///     let (dst_iter, position) = builder.iter_position();
67    ///     for dst in dst_iter {
68    ///         dst.write(make_some_struct());
69    ///         // MUST be done AFTER ownership of the value has been given to `dst.write`
70    ///         *position += 1;
71    ///     }
72    ///     let your_array = builder.assume_init();
73    /// }
74    /// # }
75    /// ```
76    #[inline(always)]
77    pub unsafe fn iter_position(
78        &'_ mut self,
79    ) -> (slice::IterMut<'_, MaybeUninit<T>>, &'_ mut usize) {
80        (self.array.iter_mut(), &mut self.position)
81    }
82
83    /// When done writing (assuming all elements have been written to),
84    /// get the inner array.
85    ///
86    /// This method is `const` since Rust 1.83.0, but non-`const` before.
87    #[rustversion::attr(since(1.83), const)]
88    #[inline(always)]
89    pub unsafe fn assume_init(self) -> GenericArray<T, N> {
90        debug_assert!(self.is_full());
91
92        let array = ptr::read(&self.array);
93        mem::forget(self);
94        GenericArray::assume_init(array)
95    }
96}
97
98impl<T, N: ArrayLength> Drop for ArrayBuilder<T, N> {
99    fn drop(&mut self) {
100        unsafe {
101            ptr::drop_in_place(
102                // Same cast as MaybeUninit::slice_assume_init_mut
103                self.array.get_unchecked_mut(..self.position) as *mut [MaybeUninit<T>]
104                    as *mut [T],
105            );
106        }
107    }
108}

110/// Similar to [`ArrayBuilder`] but uses a reference to a pre-allocated array, be
111/// it on the stack or heap.
112pub struct IntrusiveArrayBuilder<'a, T, N: ArrayLength> {
113    array: &'a mut GenericArray<MaybeUninit<T>, N>,
114    position: usize,
115}

117impl<'a, T, N: ArrayLength> IntrusiveArrayBuilder<'a, T, N> {
118    /// Begin building an array
119    ///
120    /// This method is `const` since Rust 1.83.0, but non-`const` before.
121    #[rustversion::attr(since(1.83), const)]
122    #[inline(always)]
123    pub fn new(array: &'a mut GenericArray<MaybeUninit<T>, N>) -> IntrusiveArrayBuilder<'a, T, N> {
124        IntrusiveArrayBuilder { array, position: 0 }
125    }
126
127    /// Consume an iterator, `.zip`-ing it to fill some or all of the array. This does not check if the
128    /// iterator had extra elements or too few elements.
129    ///
130    /// This makes no attempt to continue where a previous `extend` leaves off. Therefore, it should
131    /// only be used once per `ArrayBuilder`.
132    #[inline(always)]
133    pub unsafe fn extend(&mut self, source: impl Iterator<Item = T>) {
134        let (destination, position) = (self.array.iter_mut(), &mut self.position);
135
136        destination.zip(source).for_each(|(dst, src)| {
137            dst.write(src);
138            *position += 1;
139        });
140    }
141
142    /// Returns true if the write position equals the array size
143    #[inline(always)]
144    pub const fn is_full(&self) -> bool {
145        self.position == N::USIZE
146    }
147
148    /// Creates a mutable iterator for writing to the array elements.
149    ///
150    /// You MUST increment the position value (given as a mutable reference) as you iterate
151    /// to mark how many elements have been created.
152    ///
153    /// ```
154    /// #[cfg(feature = "internals")]
155    /// # {
156    /// # use generic_array::{GenericArray, internals::IntrusiveArrayBuilder, typenum::U5};
157    /// # struct SomeType;
158    /// fn make_some_struct() -> SomeType { SomeType }
159    /// unsafe {
160    ///     let mut array = GenericArray::uninit();
161    ///     let mut builder = IntrusiveArrayBuilder::<SomeType, U5>::new(&mut array);
162    ///     let (dst_iter, position) = builder.iter_position();
163    ///     for dst in dst_iter {
164    ///         dst.write(make_some_struct());
165    ///         // MUST be done AFTER ownership of the value has been given to `dst.write`
166    ///         *position += 1;
167    ///     }
168    ///     let your_array = { builder.finish(); IntrusiveArrayBuilder::array_assume_init(array) };
169    /// }
170    /// # }
171    /// ```
172    #[inline(always)]
173    pub unsafe fn iter_position(
174        &'_ mut self,
175    ) -> (slice::IterMut<'_, MaybeUninit<T>>, &'_ mut usize) {
176        (self.array.iter_mut(), &mut self.position)
177    }
178
179    /// When done writing (assuming all elements have been written to),
180    /// get the inner array.
181    #[inline(always)]
182    pub const unsafe fn finish(self) {
183        debug_assert!(self.is_full());
184        mem::forget(self)
185    }
186
187    /// Similar to [`GenericArray::assume_init`] but not `const` and optimizes better.
188    #[inline(always)]
189    pub unsafe fn array_assume_init(array: GenericArray<MaybeUninit<T>, N>) -> GenericArray<T, N> {
190        ptr::read(&array as *const _ as *const MaybeUninit<GenericArray<T, N>>).assume_init()
191    }
192}

206/// **UNSAFE**: Consumes an array one element at a time.
207///
208/// You MUST increment the position while iterating and any leftover elements
209/// will be dropped if position does not go to N
210pub struct ArrayConsumer<T, N: ArrayLength> {
211    array: ManuallyDrop<GenericArray<T, N>>,
212    position: usize,
213}
214
215impl<T, N: ArrayLength> ArrayConsumer<T, N> {
216    /// Give ownership of the array to the consumer
217    #[inline(always)]
218    pub const fn new(array: GenericArray<T, N>) -> ArrayConsumer<T, N> {
219        ArrayConsumer {
220            array: ManuallyDrop::new(array),
221            position: 0,
222        }
223    }
224
225    /// Creates an iterator and mutable reference to the internal position
226    /// to keep track of consumed elements.
227    ///
228    /// You MUST increment the position as you iterate to mark off consumed elements.
229    #[inline(always)]
230    pub unsafe fn iter_position(&'_ mut self) -> (slice::Iter<'_, T>, &'_ mut usize) {
231        (self.array.iter(), &mut self.position)
232    }
233}
234
235impl<T, N: ArrayLength> Drop for ArrayConsumer<T, N> {
236    fn drop(&mut self) {
237        unsafe {
238            ptr::drop_in_place(self.array.get_unchecked_mut(self.position..));
239        }
240    }
